def convert(answer)

    convert_hash(answer)
  when Array
    convert_array(answer)
  else
    answer
  end
end

def handle_answer(answer)
  return nil  if answer.nil?
  answer = convert(answer)
  field = opts[:field]
  if field
    answer = answer[field]
  end
  answer
end
