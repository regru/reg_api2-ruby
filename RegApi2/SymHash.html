def method_missing(key, *args, &block)
  if key.to_s =~ /\A(.+)=\z/
    self[$1] = args.first
    return args.first
  end
  if key.to_s =~ /\A(.+)\?\z/
    return !!self[$1]
  end
  return self[key]  if has_key?(key)
  nil
end

def self.from(source)
  case source
  when Hash
    res = SymHash.new
    source.each_pair do |key, value|
      res[key] = self.from(value)
    end
    res
  when Array
    source.map do |el|
      self.from(el)
    end
  else
    source
  end
end
